TABLE office {
  id UUID [PRIMARY KEY]
  name VARCHAR
  address VARCHAR
  start_time TIME
  finish_time TIME
}

TABLE master {
  id UUID [PRIMARY KEY]
  name VARCHAR
  seniority DATE 
  StartTime TIME
  FinishTime TIME
  office_id UUID
}

TABLE client {
  id UUID [PRIMARY KEY]
  name VARCHAR
  born_date DATE 
  office_id UUID
}

TABLE schedule {
  id UUID [PRIMARY KEY]
  what_date DATE 
  StartTime TIME
  FinishTime TIME
  master_id UUID
  office_id UUID
}

TABLE service {
  id UUID [PRIMARY KEY]
  name VARCHAR
  price MONEY
  duration TIME
  master_id UUID
  office_id UUID
}

TABLE appointment {
  id UUID [PRIMARY KEY]
  client_id UUID
  master_id UUID
  service_id UUID
  appointment_date DATE
  appointment_time TIME
  office_id UUID
}
